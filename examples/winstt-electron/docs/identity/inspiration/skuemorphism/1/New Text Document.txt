<script>
  // Build clock ticks
  const clock = document.getElementById('clock');
  for (let i = 0; i < 60; i++) {
    const t = document.createElement('div');
    t.className = 'tick' + (i % 5 === 0 ? ' major' : '');
    t.style.transform = `rotate(${i * 6}deg)`;
    clock.appendChild(t);
  }

  // Pin hands at ~10:10 like the screenshot
  document.getElementById('hourHand').style.transform = 'rotate(305deg)';
  document.getElementById('minHand').style.transform  = 'rotate(60deg)';

  // Helpers
  const fmt = s => [s/3600|0, s/60%60|0, s%60].map(n => String(n).padStart(2,'0')).join(':');

  let total   = 3*3600 + 45*60 + 18;
  let session = 1*3600 + 23*60 + 45;
  let breakT  = 0*3600 + 15*60 + 30;
  let onBreak = false;
  let running = true;

  const totalEl   = document.getElementById('totalTime');
  const sessionEl = document.getElementById('sessionTime');
  const breakEl   = document.getElementById('breakTime');
  const breakBtn  = document.getElementById('breakBtn');
  const stopBtn   = document.getElementById('stopBtn');

  setInterval(() => {
    if (!running) return;
    if (onBreak) {
      breakT++;
    } else {
      total++; session++;
    }
    totalEl.textContent   = fmt(total);
    sessionEl.textContent = fmt(session);
    breakEl.textContent   = fmt(breakT);
  }, 1000);

  breakBtn.addEventListener('click', () => {
    onBreak = !onBreak;
    breakBtn.querySelector('span:last-child').textContent =
      onBreak ? 'End Break' : 'Start Break';
  });

  stopBtn.addEventListener('click', () => {
    running = !running;
    stopBtn.querySelector('span:last-child').textContent =
      running ? 'Stop Tracking' : 'Resume Tracking';
  });
</script>
