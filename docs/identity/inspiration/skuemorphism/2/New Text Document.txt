<script>
  // Link custom scrollbar to feed scroll
  const feed = document.getElementById('feed');
  const thumb = document.getElementById('thumb');
  const track = thumb.parentElement;

  function updateThumb() {
    const trackH = track.clientHeight - 2;
    const ratio = feed.clientHeight / feed.scrollHeight;
    const thumbH = Math.max(40, trackH * ratio);
    thumb.style.height = thumbH + 'px';
    const maxScroll = feed.scrollHeight - feed.clientHeight;
    const maxTop = trackH - thumbH;
    const top = maxScroll > 0 ? (feed.scrollTop / maxScroll) * maxTop : 0;
    thumb.style.top = (top + 1) + 'px';
  }

  feed.addEventListener('scroll', updateThumb);
  window.addEventListener('resize', updateThumb);
  updateThumb();

  // Drag the thumb
  let dragging = false, startY = 0, startTop = 0;
  thumb.addEventListener('mousedown', e => {
    dragging = true;
    startY = e.clientY;
    startTop = parseFloat(thumb.style.top) || 0;
    e.preventDefault();
  });
  window.addEventListener('mousemove', e => {
    if (!dragging) return;
    const dy = e.clientY - startY;
    const trackH = track.clientHeight - 2;
    const thumbH = thumb.clientHeight;
    const maxTop = trackH - thumbH;
    const newTop = Math.max(0, Math.min(maxTop, startTop + dy));
    const ratio = newTop / maxTop;
    feed.scrollTop = ratio * (feed.scrollHeight - feed.clientHeight);
  });
  window.addEventListener('mouseup', () => dragging = false);
</script>
